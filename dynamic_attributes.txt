#[component]
fn App() -> impl IntoView {
    let (count, set_count) = signal(0);

    view! {
        <button
            on:click=move |_| {
                *set_count.write() += 1;
            }
            // the class: syntax reactively updates a single class
            // here, the `red` class is set when `count` is odd
            class:red=move || count.get() % 2 == 1

            // add following to index.html/css file
            //<style>
            //  .red {
            //      color: red;
            //  }
            //</style>
        >
            "Click me: "
            {count}
        </button>
    }
}

// use following syntax for classes with mix of -, numbers and other characters
class=("button-20", move || count.get() % 2 == 1)

// tuple syntax also supports multiple class assignments
class=(["button-20", "rounded"], move || count.get() % 2 == 1)


//Individual CSS properties can be directly updated
let (count, set_count) = signal(0);

view! {
    <button
        on:click=move |_| {
            *set_count.write() += 10;
        }
        // set the `style` attribute
        style="position: absolute"
        // and toggle individual CSS properties with `style:`
        style:left=move || format!("{}px", count.get() + 100)
        style:background-color=move || format!("rgb({}, {}, 100)", count.get(), 100)
        style:max-width="400px"
        // Set a CSS variable for stylesheet use
        style=("--columns", move || count.get().to_string())
    >
        "Click to Move"
    </button>
}

//

<progress
    max="50"
    // signals are functions, so `value=count` and `value=move || count.get()`
    // are interchangeable.
    value=count
/>

// derived signal to re-use result/variable/value (a closure that accesses a signal)
// alows for creation of reactive computed values that can be used in multiple places in applications with minimal overhead.

let double_count = move || count.get() * 2;

<progress
    max="50"
    value=double_count
/>
<p>
    "Double Count: "
    {double_count}
</p>

// view!{} macro provides additional inner_html that over-rides any children given to the element
// Note:: not ESCAPED! - trusted input or manual pass required
let html = "<p>This HTML will be injected.</p>";
view! {
  <div inner_html=html/>
}
